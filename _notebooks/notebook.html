def _plot(model_fit):
  validation = model_fit.history['val_acc']
  train = model_fit.history['acc']
  max_val_acc = max(validation)
  epoch = validation.index(max_val_acc)
  
  plt.plot(train, 'bo', label="Training Accuracy")
  plt.plot(validation, 'b', label="Validation Accuracy")
  plt.title("Training Acc. vs Validation Acc.")
  plt.legend()
  plt.show()
  
  print(f"Max. Validation Accuracy: {max_val_acc:.3f} at epoch #{epoch}")
  print(f"Training accuracy at epoch #{epoch}: {train[epoch]:.3f}")
